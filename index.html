                                .dimension(petDimension)
                                .group(petGroup)
                                .width(350)
                                .height(300)
                                .radius(100)
                                .innerRadius(20)
                                .on('renderlet', function (chart) {
                                    chart
                                        .selectAll('text.pie-slice')
                                        .text(function (d) {
                                            return d.data.key + ' ' + dc.utils.printSingleValue((d.endAngle - d.startAngle) / (2 * Math.PI) * 100) + '%';
                                        });
                                });
